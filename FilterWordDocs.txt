$word = New-Object -ComObject Word.Application
$word.Visible = $false
$word.DisplayAlerts = 0

$filePath = "C:\your\docx\folder\test.docx"  # Replace with your actual file path
$outputPath = "C:\your\output\folder\test.html"

try {
    Write-Host "Opening..."
    $doc = $word.Documents.Open($filePath, $false, $true)
    Write-Host "Saving..."
    $doc.SaveAs($outputPath, 10)
    $doc.Close()
    Write-Host "Success!"
} catch {
    Write-Warning "Problem: $_"
}

$word.Quit()
[System.Runtime.Interopservices.Marshal]::ReleaseComObject($word) | Out-Null
[GC]::Collect(); [GC]::WaitForPendingFinalizers()
